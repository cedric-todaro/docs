\chapter{Généralités sur les fonctions}

\section{La notion de fonction}

\subsection{Définition et exemples}

\definition

\def\manote{Le concept de fonction s'est construit progressivement au XVII\textsuperscript{e} siècle avec les travaux de \textbf{René Descartes} et \textbf{Pierre de Fermat}, qui ont révolutionné les mathématiques en reliant la géométrie aux équations algébriques via le système de coordonnées. Cependant, c'est au XVIII\textsuperscript{e} siècle que \textbf{Leonhard Euler} a formalisé la notion moderne, définissant la fonction comme une relation analytique précise permettant de lier deux variables entre elles.}

Une fonction\sidenote{\manote} $f$ est un procédé qui, à tout nombre réel $x$ appartenant à un ensemble de départ $D$, associe au plus un nombre réel, noté $f(x)$.

\begin{marginfigure}\includegraphics[width=\linewidth]{img/fct/euler.jpg}\caption{Leonhard Euler}\end{marginfigure}

\begin{itemize}
	\item \textbf{La variable :} $x$ est la valeur d'entrée.
	\item \textbf{L'image :} $f(x)$ est le résultat du calcul.
	\item \textbf{L'antécédent :} $x$ est un antécédent de $f(x)$.
	\item \textbf{Ensemble de définition :} L'ensemble $D$ des valeurs de $x$ pour lesquelles $f(x)$ existe.
\end{itemize}

\rem

On note :
\[
	\begin{array}{r@{\quad:\quad}l@{\quad\to\quad}l}
		f & D & \R   \\
		  & x & f(x)
	\end{array}
\]

\ex

Soit $f(x) = 3x - 5$ définie sur $\R$.

\begin{itemize}
	\item L'image de $-2$ est $f(-2) = 3\times (-2) - 5 = -11$.
	\item Pour trouver l'antécédent de $4$ :
	      \[
		      \begin{array}{rl}
			           & f(x) = 4   \\
			      \iff & 3x - 5 = 4 \\
			      \iff & 3x = 9     \\
			      \iff & x = 3
		      \end{array}
	      \]
\end{itemize}

\ex

Soit $g(x) = 2x^2 - 3x + 1$ définie sur $\R$.

\begin{itemize}
	\item \textbf{Calcul d'image :}\\ Pour $x = 4$ :
	      \[ g(4) = 2\times (4)^2 - 3\times (4) + 1 \quad = 2(16) - 12 + 1 \quad = 21 \]
	\item \textbf{Recherche d'antécédent de $1$ :}
	      $$
		      \def\arraystretch{1.3}
		      \begin{array}{rl}
			           & g(x) = 1                               \\
			      \iff & 2x^2 - 3x + 1 = 1                      \\
			      \iff & 2x^2 - 3x = 0                          \\
			      \iff & x(2x - 3) = 0                          \\
			      \iff & \begin{cases}x_1=0\\x_2=1,5\end{cases}
		      \end{array}
	      $$
	      Les antécédents sont $0$ et $1,5$.
\end{itemize}


\subsection{Courbe représentative}

\definition

La courbe représentative $\Cf$ est l'ensemble des points $M(x; y)$ tels que $y = f(x)$.

\methode Représentation graphique d'une fonction

Soit la fonction $f$ définie sur $[-2, 3]$ par $f(x) = 0.5x^2 - x - 1$.

\begin{itemize}
	\item Tableau de valeurs
	      Pour construire la courbe, on calcule d'abord quelques images\sidenote{Nous pouvons utiliser une calculatrice en mode tableur pour automatiser les calculs\\ \cimg{\linewidth}{img/fct/calc.png}} :
	      \begin{itemize}
		      \item $f(-2) = 0.5\times (-2)^2 - (-2) - 1 \quad = 3$
		      \item $f(-1) = 0.5\times (-1)^2 - (-1) - 1 \quad = 0,5$
		      \item $f(0) = \ldots$
	      \end{itemize}

	      $$
		      \def\arraystretch{1.5}
		      \begin{array}{|c|c|c|c|c|c|c|} \hline
			      x    & -2 & -1  & 0  & 1    & 2  & 3   \\ \hline
			      f(x) & 3  & 0,5 & -1 & -1,5 & -1 & 0,5 \\ \hline
		      \end{array}
	      $$

	\item Courbe représentative $\Cf$

	      On place les points de coordonnées $\pa{x\ ;\ f(x)}$ dans un repère.

	      \begin{center}
		      \begin{tikzpicture}
			      \begin{axis}[ axis lines=middle, xmin=-3, xmax=4, ymin=-3, ymax=4, xlabel=$x$, ylabel=$y$, grid=major, width=8cm, height=8cm ]
				      \node[font=\large, fill=white] at (axis cs: -2, 1.5) {\Cf};
				      \addplot[domain=-2:3, samples=100, blue, thick]{0.5*x^2 - x - 1};
				      \addplot[ only marks, mark=*, mark options={color=red, fill=red}, ]
				      coordinates { (-2, 3) (-1, 0.5) (0, -1) (1, -1.5) (2, -1) (3, 0.5) };
			      \end{axis}
		      \end{tikzpicture}
	      \end{center}
\end{itemize}

\methode Effectuer une lecture graphique

\begin{enumerate}
	\item \textbf{Lire une image\sidenote{~\\ \tikza} :} Partir de l'axe des abscisses ($x$), monter/descendre jusqu'à la courbe, puis lire l'ordonnée ($y$).
	\item \textbf{Lire un antécédent\sidenote{~\\ \tikzb} :} Partir de l'axe des ordonnées ($y$), tracer une horizontale jusqu'à la courbe, puis lire l'abscisse ($x$) du point d'intersection.
\end{enumerate}
