\begin{description}

    \item[$x$] $\rightarrow~1$
    \item[$x^2$] $\rightarrow~2x$
    \item[$x^3$] $\rightarrow~3x^2$
    \item[$x^n$] $\rightarrow~nx^{n-1}$
    \item[$\frac{1}{x}$] $\rightarrow~\frac{-1}{x^2}$
    \item[$\sqrt{x}$] $\rightarrow~\frac{1}{2\sqrt{x}}$
    \item[$e^x$] $\rightarrow~e^x$
\end{description}

    \item[$e^u$] $\rightarrow~u'e^u$
\end{description}
