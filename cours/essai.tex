\begin{defi}[Espérance mathématique]
	L'espérance d'une variable aléatoire discrète est définie par :
	$$\esp(X) = \sum_i x_i \prob(X = x_i)$$
\end{defi}

\begin{rem}
	Pour un dé équilibré :
	$\esp(X) = \dfrac{1+2+3+4+5+6}{6} = 3.5$
\end{rem}

\section{Relation de récurrence (Rappel)}

\begin{ex}
	\begin{enumerate}
		\item[a)] Considérons la suite $(u_n)$ où l'on passe d'un terme au suivant en ajoutant $5$.
		      Si le premier terme est égal à $3$, les termes suivants sont :
		      $$
			      u_0 = 3, \quad u_1 = 8, \quad u_2 = 13, \quad u_3 = 18.
		      $$
		      Une telle suite est appelée une \textbf{suite arithmétique} de raison $5$ et de premier terme $3$.
		      La suite est donc définie par :
		      $$
			      \begin{cases}
				      u_0 = 3 \\
				      u_{n+1} = u_n + 5
			      \end{cases}
		      $$

		\item[b)] Soit la suite arithmétique $(v_n)$ de premier terme $5$ et de raison $-2$.
		      Les premiers termes successifs sont :
		      $$
			      v_0 = 5, \quad v_1 = 3, \quad v_2 = 1, \quad v_3 = -1.
		      $$
		      La suite est donc définie par :
		      $$
			      \begin{cases}
				      v_0 = 5 \\
				      v_{n+1} = v_n - 2
			      \end{cases}
		      $$
	\end{enumerate}
\end{ex}

\begin{defi}
	Une suite $(u_n)$ est une \textbf{suite arithmétique} s'il existe un nombre $r$ tel que :
	$$
		u_{n+1} = u_n + r.
	$$
	Le nombre $r$ est appelé \textbf{raison} de la suite.
\end{defi}

\section{Forme explicite en fonction de $n$}

\begin{meth}[Exprimer une suite arithmétique en fonction de $n$]

	Pour préparer une course, un athlète décide de s'entraîner de façon progressive.

	Il commence son entraînement au « jour~0 » par un footing de $3\,000$~m.

	Au « jour~1 », il court $3\,150$~m. Au « jour~2 », il court $3\,300$~m,

	puis ainsi de suite en parcourant chaque jour $150$~m de plus que la veille.

	On note $u_n$ la distance parcourue au « jour~$n$ » d'entraînement.

	\begin{enumerate}
		\item[a)] Calculer $u_3$ et $u_4$.
		\item[b)] Quelle est la nature de la suite $(u_n)$ ? On donnera son premier terme et sa raison.
		\item[c)] Exprimer $u_{n+1}$ en fonction de $u_n$.
		\item[d)] Exprimer $u_n$ en fonction de $n$.
	\end{enumerate}

	\begin{correction}
		\begin{enumerate}
			\item[a)] $u_0 = 3\,000$, $u_1 = 3\,150$, $u_2 = 3\,300$, $u_3 = 3\,450$, $u_4 = 3\,600$.

			\item[b)] $(u_n)$ est une suite arithmétique de premier terme $u_0 = 3\,000$ et de raison $r = 150$.
			      On parle ici de \textbf{croissance linéaire}.

			\item[c)] $u_{n+1} = u_n + 150$.

			\item[d)] Après $1$ jour : $u_1 = 3\,000 + 150 \times 1$. \\
			      Après $2$ jours : $u_2 = 3\,000 + 150 \times 2$. \\
			      Après $3$ jours : $u_3 = 3\,000 + 150 \times 3$. \\
			      De manière générale : $u_n = 3\,000 + 150n$.
		\end{enumerate}
	\end{correction}
\end{meth}

\begin{prop}
	Si $(u_n)$ est une suite arithmétique de raison $r$ et de premier terme $u_0$, alors :
	$$
		u_n = u_0 + nr.
	$$
\end{prop}

\begin{meth}[Déterminer une expression en fonction de $n$ d'une suite arithmétique]

	\begin{enumerate}
		\item[a)] Déterminer l'expression en fonction de $n$ de la suite arithmétique définie par :
		      $$
			      \begin{cases}
				      u_0 = 7 \\
				      u_{n+1} = u_n - 4
			      \end{cases}
		      $$

		\item[b)] Déterminer l'expression en fonction de $n$ de la suite arithmétique définie par :
		      $$
			      \begin{cases}
				      u_1 = 5 \\
				      u_{n+1} = u_n + 3
			      \end{cases}
		      $$
	\end{enumerate}

	\begin{correction}
		\begin{enumerate}
			\item[a)] On a $u_0 = 7$ et $r = -4$, donc :
			      $$
				      u_n = u_0 + nr = 7 + n \times (-4) = 7 - 4n.
			      $$

			\item[b)] On a $u_1 = 5$ et $r = 3$. Le terme $u_0$ n'est pas donné, mais on peut le calculer :
			      $u_0 = u_1 - 3 = 2$. Donc :
			      $$
				      u_n = u_0 + nr = 2 + 3n.
			      $$
		\end{enumerate}
	\end{correction}

\end{meth}
\begin{rem}
	Il peut être pratique d'appliquer directement la formule : $\boxed{u_n = u_1 + (n-1)r}$
\end{rem}

\section{Sens de variation et représentation graphique (Rappel)}

\subsection{Sens de variation}

\begin{prop}
	Soit $(u_n)$ une suite arithmétique de raison $r$.
	\begin{itemize}
		\item Si $r > 0$, alors la suite $(u_n)$ est \textbf{croissante}.
		\item Si $r < 0$, alors la suite $(u_n)$ est \textbf{décroissante}.
	\end{itemize}
\end{prop}

\begin{meth}[Déterminer le sens de variation d'une suite arithmétique]

	Étudier les variations des suites arithmétiques $(u_n)$ et $(v_n)$ définies par :
	\begin{enumerate}
		\item[a)] $u_n = 3 + 5n$
		\item[b)] $\begin{cases} v_0 = -3 \\ v_{n+1} = v_n - 4 \end{cases}$
	\end{enumerate}

	\begin{correction}
		\begin{enumerate}
			\item[a)] $(u_n)$ est croissante car de raison positive $r = 5$.
			\item[b)] On passe d'un terme au suivant en ajoutant $-4$. $(v_n)$ est décroissante car de raison négative $r = -4$.
		\end{enumerate}
	\end{correction}
\end{meth}

\subsection{Représentation graphique}

Les points de la représentation graphique d'une suite arithmétique sont \textbf{alignés}.

\section{Somme des termes d'une suite arithmétique}

\begin{prop}
	La somme de termes consécutifs d'une suite arithmétique est égale au nombre de termes multiplié par la moyenne du premier et du dernier terme :
	$$
		u_p + u_{p+1} + \cdots + u_q = (q - p + 1) \times \frac{u_p + u_q}{2}.
	$$
	En notation sigma :
	$$
		\sum_{k=p}^{q} u_k = (q - p + 1) \times \frac{u_p + u_q}{2}.
	$$
\end{prop}

\begin{meth}[Calculer la somme des termes d'une suite arithmétique]

	On reprend le contexte de la méthode de la partie~2 : $u_n = 3\,000 + 150n$.

	\begin{enumerate}
		\item[a)] Quelle distance aura-t-il parcourue au total lorsqu'il sera au « jour~15 » ?
		\item[b)] Quelle distance aura-t-il parcourue au total entre le « jour~8 » et le « jour~12 » ?
	\end{enumerate}

	\begin{correction}
		\begin{enumerate}
			\item[a)] La somme $S = u_0 + u_1 + \cdots + u_{15}$ comporte $16$ termes. Ainsi :
			      $$
				      S = 16 \times \frac{u_0 + u_{15}}{2} = 16 \times \frac{3\,000 + (3\,000 + 150 \times 15)}{2}
				      = 16 \times \frac{8\,250}{2} = 66\,000.
			      $$
			      $$
				      \sum_{k=0}^{15} u_k = 66\,000 \text{ m}.
			      $$
			      L'athlète a parcouru $66\,000$~m soit $66$~km au « jour~15 ».

			\item[b)] La somme $u_8 + u_9 + u_{10} + u_{11} + u_{12}$ comporte $5$ termes. Ainsi :
			      $$
				      \sum_{k=8}^{12} u_k = 5 \times \frac{u_8 + u_{12}}{2}
				      = 5 \times \frac{(3\,000 + 150 \times 8) + (3\,000 + 150 \times 12)}{2}
				      = 5 \times \frac{9\,000}{2} = 22\,500.
			      $$
			      L'athlète a parcouru $22\,500$~m soit $22{,}5$~km entre le « jour~8 » et le « jour~12 ».
		\end{enumerate}
	\end{correction}
\end{meth}

\section{Moyenne arithmétique de deux nombres}

\begin{defi}
	La \textbf{moyenne arithmétique} d'une liste de nombres est la somme des valeurs divisée par le nombre de valeurs.
\end{defi}

\begin{meth}[Calculer une moyenne arithmétique]

	\begin{enumerate}
		\item[a)] Calculer la moyenne arithmétique des nombres $-3$ et $19$.
		\item[b)] Peut-on affirmer que chaque terme d'une suite arithmétique est la moyenne arithmétique du terme qui le précède et du terme qui le suit ?
	\end{enumerate}

	\begin{correction}
		\begin{enumerate}
			\item[a)]
			      $$
				      m = \frac{-3 + 19}{2} = \frac{16}{2} = 8.
			      $$

			\item[b)] Soit $(u_n)$ une suite arithmétique de raison $r$. On a $u_{n+1} = u_n + r$ et $u_{n-1} = u_n - r$. Alors :
			      $$
				      \frac{u_{n-1} + u_{n+1}}{2} = \frac{(u_n - r) + (u_n + r)}{2} = \frac{2u_n}{2} = u_n.
			      $$
			      Donc $u_n$ est bien la moyenne arithmétique du terme qui le précède et du terme qui le suit.
		\end{enumerate}
	\end{correction}
\end{meth}

\section*{Résumé}

Soit $(u_n)$ une suite arithmétique de raison $r$ et de premier terme $u_0$.

\begin{description}
	\item[Définition (récurrence)] $u_{n+1} = u_n + r$.
	\item[Forme explicite] $u_n = u_0 + nr$.
	\item[Variations] Si $r > 0$ : $(u_n)$ croissante. Si $r < 0$ : $(u_n)$ décroissante.
	\item[Somme de termes consécutifs]
	      $$
		      \sum_{k=p}^{q} u_k = (q-p+1) \times \frac{u_p + u_q}{2}.
	      $$
	\item[Représentation graphique] Les points sont alignés (croissance linéaire).
	\item[Propriété de moyenne] Chaque terme est la moyenne arithmétique de ses voisins :
	      $u_n = \dfrac{u_{n-1} + u_{n+1}}{2}$.
\end{description}
