\section{Expériences aléatoires et univers}
% ══════════════════════════════════════════════

\begin{defin}{Expérience aléatoire}
	Une \textbf{expérience aléatoire} est une expérience dont on ne peut pas
	prévoir le résultat à l'avance, mais dont on connaît tous les résultats
	possibles. L'ensemble de tous ces résultats possibles s'appelle
	l'\textbf{univers} et se note $\Omega$.
\end{defin}

\begin{ex}{Lancer de dé}
	On lance un dé à six faces équilibré.
	\begin{itemize}
		\item L'univers est $\Omega = \{1, 2, 3, 4, 5, 6\}$.
		\item Obtenir le résultat $3$ est une \textbf{issue}.
	\end{itemize}
\end{ex}

\begin{rem}
	Le résultat d'une expérience aléatoire ne peut pas être prévu à l'avance,
	même en connaissant parfaitement les conditions initiales.
\end{rem}

\begin{defin}{Événement}
	Un \textbf{événement} est une partie de $\Omega$. On distingue :
	\begin{itemize}
		\item l'\textbf{événement certain} $\Omega$ (toujours réalisé) ;
		\item l'\textbf{événement impossible} $\emptyset$ (jamais réalisé) ;
		\item un \textbf{événement élémentaire} $\{\omega\}$ (une seule issue).
	\end{itemize}
\end{defin}

\begin{ex}{Événements sur un dé}
	Soit $A = \{2, 4, 6\}$ l'événement « obtenir un nombre pair » et
	$B = \{5, 6\}$ l'événement « obtenir un nombre supérieur à 4 ».
	\begin{itemize}
		\item $A \cup B = \{2, 4, 5, 6\}$ : « pair \textbf{ou} supérieur à 4 ».
		\item $A \cap B = \{6\}$ : « pair \textbf{et} supérieur à 4 ».
		\item $\bar{A} = \{1, 3, 5\}$ : « ne pas obtenir un nombre pair ».
	\end{itemize}
\end{ex}

\begin{attention}
	$A$ et $B$ sont dits \textbf{incompatibles} si $A \cap B = \emptyset$.
	Ne pas confondre avec des événements indépendants — ce sont deux notions
	distinctes.
\end{attention}

\section{Probabilités}
% ══════════════════════════════════════════════

\begin{defin}{Probabilité}
	Une \textbf{probabilité} sur $\Omega$ est une fonction $P$ qui associe à
	chaque événement $A$ un réel $P(A)$ vérifiant :
	\begin{enumerate}
		\item $0 \leq P(A) \leq 1$ ;
		\item $P(\Omega) = 1$ ;
		\item si $A \cap B = \emptyset$ alors $P(A \cup B) = P(A) + P(B)$.
	\end{enumerate}
\end{defin}

\begin{prop}{Probabilité du complémentaire}
	Pour tout événement $A$ :
	\[
		P(\bar{A}) = 1 - P(A).
	\]
\end{prop}

\begin{preuve}{}
	$A$ et $\bar{A}$ sont incompatibles et $A \cup \bar{A} = \Omega$, donc
	par le troisième axiome :
	\[
		P(A) + P(\bar{A}) = P(\Omega) = 1,
	\]
	d'où $P(\bar{A}) = 1 - P(A)$.
\end{preuve}

\begin{theo}{Formule de la réunion}
	Pour tous événements $A$ et $B$ :
	\[
		P(A \cup B) = P(A) + P(B) - P(A \cap B).
	\]
\end{theo}

\begin{demo}{}
	On décompose $A \cup B$ en trois parties disjointes :
	\[
		A \cup B = (A \setminus B) \;\cup\; (A \cap B) \;\cup\; (B \setminus A).
	\]
	Par additivité :
	$P(A \cup B) = P(A\setminus B) + P(A \cap B) + P(B \setminus A)$.
	Puisque $P(A) = P(A\setminus B) + P(A\cap B)$ et
	$P(B) = P(B\setminus A) + P(A\cap B)$, on obtient le résultat.
\end{demo}

\begin{ex}{Application de la formule}
	Dans une classe de 30 élèves, 18 font du sport, 12 font de la musique et
	6 font les deux. Quelle est la probabilité qu'un élève choisi au hasard
	fasse du sport ou de la musique ?
	\[
		P(S \cup M) = \frac{18}{30} + \frac{12}{30} - \frac{6}{30}
		= \frac{24}{30} = \frac{4}{5}.
	\]
\end{ex}

\section{Équiprobabilité}
% ══════════════════════════════════════════════

\begin{defin}{Situation d'équiprobabilité}
	Lorsque tous les événements élémentaires ont la même probabilité, on dit
	que la loi est \textbf{équiprobable}. Si $|\Omega| = n$, alors pour tout
	événement $A$ :
	\[
		P(A) = \frac{|A|}{|\Omega|} = \frac{\text{nombre d'issues favorables}}%
		{\text{nombre total d'issues}}.
	\]
\end{defin}

\begin{rem}
	L'équiprobabilité est valable pour un dé non pipé, une pièce équilibrée,
	un tirage au sort dans une urne homogène, etc.
\end{rem}

\begin{ex}{Tirage de carte}
	On tire une carte au hasard dans un jeu de 52 cartes.
	\begin{itemize}
		\item $P(\text{as}) = \dfrac{4}{52} = \dfrac{1}{13} \approx 7{,}7\,\%$.
		\item $P(\text{coeur}) = \dfrac{13}{52} = \dfrac{1}{4} = 25\,\%$.
		\item $P(\text{as de coeur}) = \dfrac{1}{52} \approx 1{,}9\,\%$.
		\item $P(\text{as ou coeur}) = \dfrac{4 + 13 - 1}{52}
			      = \dfrac{16}{52} = \dfrac{4}{13}$.
	\end{itemize}
\end{ex}

\begin{attention}
	L'équiprobabilité doit être \textbf{justifiée} par le contexte (dé
	équilibré, tirage au sort…). On ne peut pas la supposer sans raison.
\end{attention}

\section{Fréquences et probabilités}
% ══════════════════════════════════════════════

\begin{defin}{Fréquence}
	Lors de $n$ répétitions d'une expérience aléatoire, si un événement $A$
	est réalisé $k$ fois, sa \textbf{fréquence d'apparition} est :
	\[
		f = \frac{k}{n}.
	\]
\end{defin}

\begin{theo}{Loi des grands nombres (admis)}
	Lorsque le nombre $n$ de répétitions d'une expérience aléatoire augmente
	indéfiniment, la fréquence d'apparition d'un événement $A$ se rapproche
	de sa probabilité $P(A)$ :
	\[
		\frac{k}{n} \xrightarrow[n \to +\infty]{} P(A).
	\]
\end{theo}

\begin{rem}
	Ce théorème justifie l'approche fréquentiste : on peut \emph{estimer}
	une probabilité inconnue en répétant l'expérience un grand nombre de fois.
	Il ne garantit pas que la fréquence \emph{égale} la probabilité pour un
	$n$ fini.
\end{rem}

\subsection{En python}
\begin{ex}{Simulation}
	On lance une pièce 1000 fois et on obtient 487 fois pile. La fréquence
	est $\frac{487}{1000} = 0{,}487$, proche de la probabilité théorique
	$P(\text{pile}) = 0{,}5$.

	\begin{lstlisting}[language=Python]
import random

n = 1000
pile = 0

for i in range(n):
    if random.randint(0, 1) == 1:
        pile = pile + 1

frequence = pile / n
print(frequence)
\end{lstlisting}

\end{ex}
